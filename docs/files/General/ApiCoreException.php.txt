namespace Cloudonix\Common;

class ApiCoreException extends Exception
{

	public function onErrorNullValue() {
		$errorMsg = 'Provided values can not by NULL';
		return $errorMsg;
	}

	public function onErrorHttpConnect() {
		$errorMsg = 'Remote Api.Core API server is unreachable';
		return $errorMsg;
	}

}
