namespace Cloudonix\ApiCore;

/**
 * Dnid Class - Cloudonix Api.Core Dnid Data Model API
 *
 * A Dnid represents a logical unit, normally attached to a specific domain. When a call is intercepted by Cloudonix,
 * either from a remote UA (User Agent) or from a remote carrier - it will be parsed and governed, in accordance to the
 * Dnid provided. The Dnid can be either a specific number, or a regular expression. Additional information can be
 * obtained from https://cloudonix.io/wp-content/uploads/2018/12/Cloudonix-Platform-Administrator-Guide.pdf
 *
 * @package Cloudonix\ApiCore
 */
class Dnid
{
	/** @var ApiCoreClient $apiClientConnector The previously initiated ApiCoreClient object */
	public $apiClientConnector = null;

	/** @var integer $tenantId The tenant ID initiating the API, for system privilage - null */
	public $tenantId = null;

	/**
	 * Dnid constructor.
	 * @param ApiCoreClient $connector Cloudonix ApiCoreClient object
	 */
	public function __construct($connector = null)
	{
		try {

			if ($connector == null) {
				throw new \Exception("Input values can not be null");
			}

			$this->apiClientConnector = $connector->apiClientConnector;
			$this->tenantId = $connector->tenantId;

		} catch (\Exception $e) {
			die("Exception Raised at " . $e->getFile() . " Line: " . $e->getLine() . " Message: [" . $e->getMessage() . "]");
		}
	}

	/**
	 * Dnid destructor
	 */

	}

}
